def add(value)
  node = Node.new(value, nil, nil)
  if @head == nil
    @head = node
  end
  if @tail != nil
    @tail.nex = node
    node.prev = @tail
  end
  @tail = node
  return self
end

def pop
  if @tail != nil
    if @tail.prev != nil
      toReturn = @tail
      @tail = @tail.prev
      @tail.nex = nil
      return toReturn
    else
      lastElement = @tail
      @tail = nil
      return lastElement
    end
  else
    return "List is empty" 
  end
end

def to_array
  resultArray = []
  current = @head
  while current != nil do
    resultArray << current
    current = current.nex
  end
  return resultArray
end
